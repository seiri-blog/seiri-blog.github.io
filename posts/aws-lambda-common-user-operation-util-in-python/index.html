def create_random_password(number_digits):
    """ãƒ¦ãƒ¼ã‚¶ç™»éŒ²ç”¨ã®ãƒ©ãƒ³ãƒ€ãƒ ãƒ‘ã‚¹ãƒ¯ãƒ¼ãƒ‰ã‚’ç”Ÿæˆã™ã‚‹ã€‚
    Args:
        number_digits:ãƒ‘ã‚¹ãƒ¯ãƒ¼ãƒ‰æ¡æ•°
    Returns:
        ç”Ÿæˆã•ã‚ŒãŸãƒ©ãƒ³ãƒ€ãƒ ãƒ‘ã‚¹ãƒ¯ãƒ¼ãƒ‰ï¼ˆå¤§æ–‡å­—è‹±å­—ã€å°æ–‡å­—è‹±å­—ã€æ•°å­—ã‚’å¿…ãšå«ã‚€ï¼‰
    """
    while True:
        punctuation = '^$*.[]{}()?-"!@#%&/\\,><\':;|_~`'  # ç‰¹æ®Šæ–‡å­—ã®è¦ä»¶ã«'+='ãŒå«ã¾ã‚Œã¦ã„ãªã„ãŸã‚string.punctuationã¯æœªä½¿ç”¨
        words = string.ascii_letters + string.digits + punctuation

                and any(c.isupper() for c in password)
                and any(c.islower() for c in password)
                and any((c in punctuation) for c in password)):
            break
    return password
